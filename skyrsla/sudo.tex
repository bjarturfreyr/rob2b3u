while(true)\{\\
  if(user press up):\\
  	pin1.sendsignal\\
  elif(user press left):\\
  	pin2.sendsignal\\
  elif(user press right):\\
  	pin3.sendsignal\\
  elif(user press down):\\
  	pin4.sendsignal\\
  sendcamerafeed()\\
\}\\\\
